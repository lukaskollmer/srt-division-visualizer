      $$ q_{%ITERATION_INDEX%} = \text{lookup_table}(p: %REMAINDER_APPROX%_{2}, d: %DIVISOR_APPROX%_{2}) = %QUOTIENT_GUESS% $$

      $$q = [%QUOTIENT_DIGITS%]_{%RADIX%} = %ASS_QUOTIENT_UNSHIFTED%_{10} * 2^{%INPUT_EXP_DIFF%} = %ASS_QUOTIENT%_{10}$$
